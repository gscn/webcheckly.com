
# 一、产品定位（你这个工具是什么）

### 产品名称（暂定）

**WebCheckly / WebCheck / OpsScan**

### 一句话定位

> 输入一个网址，一分钟内生成一份**可用性 + 性能 + 页面资源存活性**的技术运维检测报告。

### 目标用户

| 用户     | 使用场景      |
| ------ | --------- |
| 独立开发者  | 发布前 / 排错  |
| 运维工程师  | 站点巡检      |
| 安全测试   | 资产发现、存活探测 |
| SEO/站长 | 页面健康度     |

# 二、核心功能模块设计（MVP）

### ✅ 必须有（第一版）

1. **URL 输入检测**
2. **页面内 URL 抓取**
3. **httpx 检测**

   * 可用性（alive）
   * 状态码
   * 页面标题
   * 响应时间
   * IP / CDN / TLS
4. **流式返回结果（实时展示）**

### 🚫 明确不做（第一版）

* 登录 / 用户系统
* 数据库存储
* 历史记录

# 三、整体流程设计（非常关键）

```text
用户输入 URL
   ↓

   ↓
抓取页面 HTML

   ↓

   ↓
检测结束
   ↓

```

func NormalizeURL(raw string) (string, error) {
	u, err := url.Parse(raw)
	if err != nil {
		return "", err
	}

	if u.Scheme == "" {
		u.Scheme = "https"
	}
	return u.String(), nil
}

## 4️⃣ 页面抓取与 URL 解析 `services/crawler.go`

```go
package services

import (
	"net/http"
	"net/url"

	"github.com/PuerkitoBio/goquery"
)

func ExtractPageUrls(target string) ([]string, error) {
	resp, err := http.Get(target)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	base, _ := url.Parse(target)
	doc, err := goquery.NewDocumentFromReader(resp.Body)
	if err != nil {
		return nil, err
	}

	urls := make(map[string]bool)

	doc.Find("a[href]").Each(func(i int, s *goquery.Selection) {
		href, _ := s.Attr("href")
		u, err := url.Parse(href)
		if err != nil {
			return
		}
		abs := base.ResolveReference(u)
		if abs.Scheme == "http" || abs.Scheme == "https" {
			urls[abs.String()] = true
		}
	})

	result := []string{}
	for u := range urls {
		result = append(result, u)
	}
	return result, nil
}
```

## 5️⃣ httpx 调用封装（核心）`services/httpx.go`

```go
package services

import (
	"bufio"
	"encoding/json"
	"os/exec"

	"opsscan/models"
)

func RunHttpx(urls []string, output chan models.HttpxResult) error {
	cmd := exec.Command(
		"httpx",
		"-json",
		"-status-code",
		"-title",
		"-response-time",
		"-ip",
		"-cdn",
		"-tls",
	)

	stdin, _ := cmd.StdinPipe()
	stdout, _ := cmd.StdoutPipe()

	if err := cmd.Start(); err != nil {
		return err
	}

	// 输入 URL
	go func() {
		for _, u := range urls {
			stdin.Write([]byte(u + "\n"))
		}
		stdin.Close()
	}()

	// 读取输出
	scanner := bufio.NewScanner(stdout)
	for scanner.Scan() {
		line := scanner.Text()
		var res models.HttpxResult
		if err := json.Unmarshal([]byte(line), &res); err == nil {
			output <- res
		}
	}

	cmd.Wait()
	close(output)
	return nil
}
```

func SendSSE(c *fiber.Ctx, event string, data any) {

## 7️⃣ 扫描接口（整合所有流程）`routes/scan.go`

```go
package routes

import (
	"opsscan/models"
	"opsscan/services"
	"opsscan/utils"

	"github.com/gofiber/fiber/v2"
)

func ScanHandler(c *fiber.Ctx) error {
	raw := c.Query("url")
	target, err := utils.NormalizeURL(raw)
	if err != nil {
		return c.Status(400).SendString("Invalid URL")
	}

	c.Set("Content-Type", "text/event-stream")
	c.Set("Cache-Control", "no-cache")

	urls, err := services.ExtractPageUrls(target)
	if err != nil {
		return c.Status(500).SendString("Failed to crawl page")
	}

	results := make(chan models.HttpxResult)
	go services.RunHttpx(urls, results)

	total := len(urls)
	count := 0

	for res := range results {
		count++
		services.SendSSE(c, "result", res)
		services.SendSSE(c, "progress", fiber.Map{
			"current": count,
			"total":   total,
		})
	}

	services.SendSSE(c, "done", fiber.Map{
		"total": total,
	})

	return nil
}
```

# 五、扫描页（SSE 核心）

## `app/scan/page.tsx`

```tsx
"use client"

import { useEffect, useState } from "react"
import { ScanResult, Progress } from "@/types/scan"
import ResultTable from "@/components/ResultTable"
import ProgressBar from "@/components/ProgressBar"
import ReportSummary from "@/components/ReportSummary"
import { useSearchParams } from "next/navigation"

export default function ScanPage() {
  const params = useSearchParams()
  const url = params.get("url")

  const [results, setResults] = useState<ScanResult[]>([])
  const [progress, setProgress] = useState<Progress>({ current: 0, total: 0 })
  const [done, setDone] = useState(false)

  useEffect(() => {
    if (!url) return

    const es = new EventSource(
      `http://localhost:3000/api/scan?url=${encodeURIComponent(url)}`
    )

    es.addEventListener("result", (e) => {
      const data = JSON.parse(e.data)
      setResults((prev) => [...prev, data])
    })

    es.addEventListener("progress", (e) => {
      setProgress(JSON.parse(e.data))
    })

    es.addEventListener("done", () => {
      setDone(true)
      es.close()
    })

    return () => es.close()
  }, [url])

  return (
    <main className="p-6 space-y-6">
      <h2 className="text-xl font-semibold">Scanning: {url}</h2>

      <ProgressBar progress={progress} />

      <ResultTable results={results} />

      {done && <ReportSummary results={results} />}
    </main>
  )
}
```

import { ScanResult } from "@/types/scan"

export default function ReportSummary({ results }: { results: ScanResult[] }) {
  const total = results.length
  const alive = results.filter(r => r.status > 0).length
  const avg =
    Math.round(
      results.reduce((sum, r) => sum + (r.response_time || 0), 0) / total

import { ScanResult } from "@/types/scan"

export function buildReport(
  target: string,
  results: ScanResult[]

  const total = results.length
  const alive = results.filter(r => r.status > 0).length
  const avg =
    Math.round(
      results.reduce((s, r) => s + (r.response_time || 0), 0) / total

  return {
    target,
    scannedAt: new Date().toISOString(),
    summary: {
      total,
      alive,
      dead: total - alive,
      avgResponse: avg,
    },
    results,
  }
}

## 1️⃣ IP 限流（GoFiber Middleware）

### 目标

> 同一 IP：
>
> * 每分钟最多 **3 次扫描**
> * 超过直接拒绝

### 实现方式

使用 **Fiber 官方 limiter**

```bash
go get github.com/gofiber/fiber/v2/middleware/limiter
```

### `main.go` 增加中间件

```go
import "github.com/gofiber/fiber/v2/middleware/limiter"

app.Use(limiter.New(limiter.Config{
	Max:        3,
	Expiration: 1 * time.Minute,
	LimitReached: func(c *fiber.Ctx) error {
		return c.Status(429).SendString("Too many scan requests")
	},
}))
```

✅ 成本极低
✅ 非常有效

### 实现：IP 检测工具

#### `utils/ssrf.go`

```go
package utils

import (
	"net"
)

func IsPrivateIP(host string) bool {
	ips, err := net.LookupIP(host)
	if err != nil {
		return true
	}

	for _, ip := range ips {
		if ip.IsLoopback() || ip.IsPrivate() {
			return true
		}
	}
	return false
}
```

---

### 在 ScanHandler 中使用

```go
u, _ := url.Parse(target)
if utils.IsPrivateIP(u.Hostname()) {
	return c.Status(400).SendString("Private IP not allowed")
}
```

👉 这一条能帮你挡掉 **90% 的恶意扫描**
